/* global _, $, Module, Events, PlayState, Modules*/
/* exported LoadingIconManager */
var LoadingIconManager = Module.extend({
	initialize: function() {
		_.bindAll(this, "onPlayhead", "showIcon", "onStateChange", "checkReadyState", "positionIcon");
		var player = this.options.player,
			$el = player.$el;
		this.$loadingIcon = $("<div/>").addClass("pjs-loading-icon");
		this.positionIcon = _.once(this.positionIcon);
		$el.append(this.$loadingIcon);
		player.on(Events.STATE_CHANGE, this.onStateChange);
		player.on(Events.PLAYHEAD_UPDATE, this.onPlayhead);
	},
	onStateChange: function(event) {
		var state = event.data;
		switch (state) {
			case "waiting":
			case "loading": // fired from flash polyfill.
			case "buffering": // fired from flash polyfill.
			case PlayState.STALLED:
			case PlayState.SEEKING:
				this.logger.info("show loading icon, state:" + state);
				clearTimeout(this.stalledTimer);
				this.stalledTimer = setTimeout(this.showIcon, 500);
				break;
			case "canplay":
			case "canplaythrough":
				if (!this.hidden) {
					this.logger.info("hide icon, state:" + state);
				}
				this.hideIcon();
				break;
			default:
				break;
		}
	},
	onPlayhead: function() {
		if (!this.options.player.isSeeking()) {
			this.hideIcon();
		}
	},
	hideIcon: function() {
		this.clearTimers();
		this.hidden = true;
		this.$loadingIcon.hide();
	},
	positionIcon: function() {
		// move it to after the video element
		this.$loadingIcon.insertAfter(this.player.element);
	},
	showIcon: function() {
		this.positionIcon();
		this.$loadingIcon.show();
		this.hidden = false;
		clearInterval(this.loadingIconInterval);
		this.loadingIconInterval = setInterval(this.checkReadyState, 500);
	},
	checkReadyState: function() {
		// TODO implement canPlay in flash, however that should be done.
		// Or if it should be done.
		if (this.player.module(Modules.VIDEO).name === "Flash.Player") {
			return;
		}
		if (this.player.module(Modules.VIDEO).canPlay()) {
			this.hideIcon();
		}
	},
	clearTimers: function() {
		clearTimeout(this.stalledTimer);
		clearInterval(this.loadingIconInterval);
	},
	destroy: function() {
		this.clearTimers();
		var player = this.options.player;
		player.off(Events.STATE_CHANGE, this.onStateChange);
		player.off(Events.PLAYHEAD_UPDATE, this.onPlayhead);
	}
}, {
	NAME: "LoadingIconManager"
});
