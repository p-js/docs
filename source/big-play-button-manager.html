/* global _, Module, Events, PlayState, BigButtonPlayView*/
/* exported BigPlayButtonManager */
var BigPlayButtonManager = Module.extend({
	initialize: function() {
		// native controls and the big play button will work on safari.
		if (!this.config.chromeless() && !this.config.useNativeControls()) {
			_.bindAll(this, "onStateChange", "show");
			var player = this.options.player,
				$el = player.$el,
				button = new BigButtonPlayView({
					player: this.options.player
				});
			this.$icon = button.$el;
			this.$icon.hide();
			$el.append(this.$icon);
			player.on(Events.STATE_CHANGE, this.onStateChange);
		}
	},
	onStateChange: function(event) {
		clearTimeout(this.waitAfterBufferingEvent);
		switch (event.data) {
			case PlayState.PAUSED:
				this.show();
				break;
				// all of these can happen while paused.
			case "seeked":
			case "suspend":
			case PlayState.STALLED:
			case "waiting":
			case "buffering": // TODO this fires from flash only.
				this.waitAfterBufferingEvent = setTimeout(this.show, 500);
				break;
			default:
				this.hide();
		}
	},
	show: function() {
		clearTimeout(this.waitAfterBufferingEvent);
		if (this.player.isPaused()) {
			this.$icon.show();
		} else {
			this.$icon.hide();
		}
	},
	hide: function() {
		this.$icon.hide();
	}
}, {
	NAME: "BigPlayButtonManager"
});
