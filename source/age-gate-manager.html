/* global _, Module, require, UserManager */
/* exported AgeGateManager */
/**
 * @ignore
 * Load the age gate code, show it, and respond to the verification.
 */
var AgeGateManager = Module.extend({
	initialize: function() {
		_.bindAll(this, "onVerification", "showAgeGate");
	},
	verify: function() {
		if (this.ageGate) {
			return;
		}
		require("mtvn-player/age-gate", this.showAgeGate);
	},
	showAgeGate: function(AgeGate) {
		var ageGate = this.ageGate = new AgeGate({
			requiredAge: this.config.requiredAge(),
			isTooYoung: this.module(UserManager).hasFailedAgeVerification()
		});
		this.logger.log("show age gate, options", ageGate.options);
		ageGate.$el.css({
			position: "absolute"
		});
		this.player.$el.append(ageGate.$el);
		this.listenTo(ageGate, AgeGate.Events.IS_OK, this.onVerification);
	},
	onVerification: function(event) {
		var isOldEnough = event.data,
			player = this.options.player;
		player.module(UserManager).setAgeVerified(isOldEnough);
		this.logger.info(isOldEnough ? "verified, play video." : "too young.");
		if (isOldEnough) {
			delete this.ageGate;
			player.play();
		}
	},
	destroy: function() {
		if (this.ageGate) {
			this.ageGate.remove();
		}
	}
}, {
	NAME: "AgeGateManager"
});
