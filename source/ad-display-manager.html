/* global _, Events, Module, require, VMAPAdManager, FreeWheelManager*/
/* exported AdDisplayManager */
/**
 * @ignore
 * Manage the Ad Display.
 */
var AdDisplayManager = Module.extend({
	initialize: function() {
		if (!this.config.chromeless()) {
			_.bindAll(this, "onAd", "onPlayhead", "showAdDisplay", "onLearnMore", "onAdComplete");
			this.listenTo(this.module(VMAPAdManager), VMAPAdManager.Events.AD, this.onAd);
			this.listenTo(this.module(FreeWheelManager), FreeWheelManager.Events.METADATA, this.onAd);
			this.listenTo(this.module(FreeWheelManager), FreeWheelManager.Events.END, this.onAdComplete);
			this.player.on(Events.PLAYHEAD_UPDATE, this.onPlayhead);
		}
	},
	/**
	 * When an ad plays, show/hide the gui.
	 */
	onAd: function(event) {
		this.currentAd = event.data;
		if (_.isObject(this.currentAd)) {
			if (this.config.isVMAP()) {
				this.model = this.module(VMAPAdManager).model;
			} else {
				this.model = this.currentAd;
			}
			this.logger.log("ad, show ad display panel", this.currentAd);
			require("pjs-gui", this.showAdDisplay);
		} else {
			this.onAdComplete();
		}
	},
	onAdComplete: function() {
		if (this.gui) {
			this.logger.log("no ad, delete gui");
			this.gui.$el.remove();
			delete this.gui;
		}
	},
	/**
	 * Create the GUI if it's not there yet.
	 */
	showAdDisplay: function(GUI) {
		if (this.currentAd) {
			var AdDisplay = GUI.AdDisplay,
				options = {
					buttonLink: this.currentAd.clickThrough
				};
			if (this.currentAd.isFreeWheel && this.currentAd.isClickable) {
				options.buttonLink = AdDisplay.LEARN_MORE_EVENT_ONLY;
			}
			if (!this.gui) {
				// create the gui div
				var $el = this.player.$el;
				this.gui = new AdDisplay(options);
				this.logger.info("create ad display", options);
				$el.append(this.gui.$el);
				this.listenTo(this.gui, AdDisplay.Events.LEARN_MORE, this.onLearnMore);
			} else {
				this.logger.info("render ad display", options);
				this.gui.render(options);
			}
			this.setCountdownTime(this.player.playhead);
		}
	},
	onLearnMore: function() {
		this.trigger(AdDisplayManager.Events.LEARN_MORE, {
			type: AdDisplayManager.Events.LEARN_MORE,
			target: this
		});
	},
	/**
	 * Update the countdown message as the playhead moves.
	 */
	onPlayhead: function(event) {
		this.setCountdownTime(event.data);
	},
	setCountdownTime: function(playheadTime) {
		if (this.currentAd && this.gui) {
			var time = Math.ceil(this.model.getAdSlotTimeRemaining(playheadTime));
			if (!isNaN(time)) {
				this.gui.renderMessage(time);
			}
		}
	},
	destroy: function() {
		if (this.gui) {
			this.gui.$el.remove();
		}
		delete this.gui;
		this.stopListening();
	}
}, {
	NAME: "AdDisplayManager",
	Events: {
		LEARN_MORE: "LEARN_MORE"
	}
});
