/* global _, Backbone*/
/* exported BigButtonPlayView */
var BigButtonPlayView = Backbone.View.extend({
	className: "pjs-placeholder-button",
	initialize: function(options) {
		this.options = options;
		_.bindAll(this, "click");
	},
	events: {
		click: "click",
		touchstart: "click"
	},
	click: function(event) {
		event.preventDefault();
		this.options.player.play();
		this.$el.hide();
	}
}, {
	NAME: "BigButtonPlayView"
});
