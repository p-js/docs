/* global Module, Url */
/* exported ServicesUrlManager */
var ServicesUrlManager = Module.extend({
	initialize: function() {
		this.test = this.config.test();
	},
	getPlaceholderURL: function(uri) {
		return this.test.placeholderURL ? this.test.placeholderURL : this.getServicesURL() + "/placeholder/" + uri;
	},
	getMediaGenURL: function(url) {
		return this.test.mediaGenURL ? this.test.mediaGenURL : Url.setParameters(this.getServicesURL() + "/mediaGen/", {
			url: url
		});
	},
	getConfigURL: function(uri) {
		return this.test.configURL ? this.test.configURL : this.getServicesURL() + "/config/" + uri;
	},
	getServicesURL: function() {
		return this.test.useStaging ? ServicesUrlManager.STAGING : ServicesUrlManager.PRODUCTION;
	}
}, {
	PRODUCTION: "http://pjs-production.elasticbeanstalk.com",
	STAGING: "http://pjs-staging.elasticbeanstalk.com",
	DEV: "http://localhost:3000",
	NAME: "ServicesUrlManager"
});
