/*global Logger, Backbone, _*/
/* exported Module */
/**
 * @ignore
 * Simple base class functionality.
 * Subclass get an options and a logger.
 * An initialize method and a destroy method.
 * And Backbone.Events.
 */
var Module = function(options) {
	this.options = options || {};
	this.player = options.player;
	this.config = options.config;
	this.controller = options.controller;
	this.module = this.player.module;
	var loggerName = this.name || options.moduleId;
	loggerName = loggerName ? "." + loggerName : "";
	this.logger = new Logger("PJS.Player" + loggerName);
};
// mixin events and some default methods
_.extend(Module.prototype, Backbone.Events, {
	initialize: function() {},
	destroy: function() {
		this.stopListening();
	}
});
// copy the extend function, this has nothing to do with Backbone.Model
Module.extend = Backbone.Model.extend;
