/*global Module, _, Events, $*/
/* exported VTTManager */
var VTTManager = Module.extend({
	initialize: function() {
		if (this.config.ccEnabled()) {
			_.bindAll(this, "onMetadata");
			this.player.on(Events.METADATA, this.onMetadata);
		}
	},
	onMetadata: function(event) {
		var mediaGen = event.data.mediaGen,
			$video = $(this.player.element);
		// clear all text tracks
		$video.find("track").remove();
		if (mediaGen && mediaGen.transcript) {
			_.each(mediaGen.transcript, function(transcript) {
				if (transcript.vtt) {
					$video.append($("<track/>").attr({
						src: transcript.vtt,
						label: transcript.label,
						srclang: transcript.srclang,
						kind: transcript.kind
					}));
				}
			});
		}
	}
}, {
	NAME: "VTTManager"
});
