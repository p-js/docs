/* global $, Module*/
/* exported ActivatedVideoElementManager */
var ActivatedVideoElementManager = Module.extend({
	initialize: function() {
		// for iPhone. Activate the video element.
		var $el = this.options.player.$el,
			$video = $("<video></video>");
		if ($el && $el.find("video").length > 0) {
			$video = $el.find("video");
		}
		// Try to activate it.
		$video[0].play();
		this.el = $video[0];
		this.logger.info("try to activate video element.");
	}
}, {
	NAME: "ActivatedVideoElementManager"
});
