/* global _*/
/* exported VMAPAdModel*/
/**
 * @ignore
 * Wrap the object with some utility functions to determine which ad is playing based on a time passed.
 */
var VMAPAdModel = function(model) {
	_.extend(this, model);
	var ranges = this.ranges = [],
		endTime;
	_.each(model.adBreaks, function(adBreak) {
		var range = _.range(adBreak.startTime, adBreak.endTime);
		if (adBreak.startTime !== endTime) {
			ranges.push(range);
		} else {
			ranges[ranges.length - 1] = _.last(ranges).concat(range);
		}
		endTime = adBreak.endTime;
	});
};
VMAPAdModel.prototype = {
	/**
	 * @ignore
	 * convert something like "00:00:29.9330000+00:00" to 29.93
	 */
	rawTime: function(seconds) {
		var b = seconds.split(/\D/);
		return (+b[0]) * 60 * 60 + (+b[1]) * 60 + (+b[2]) + (b[3] ? parseFloat("." + b[3]) : 0);
	},
	/**
	 * @param {Number} the time to evaluate agains the VMAPParser.
	 * @return the ad that's playing for the current time.
	 */
	getAd: function(currentTime) {
		if (_.isString(currentTime)) {
			currentTime = this.rawTime(currentTime);
		}
		var ad;
		_.some(this.adBreaks, function(roll) {
			if (currentTime >= roll.startTime && currentTime <= roll.endTime) {
				ad = roll;
				return true;
			}
		});
		return ad;
	},
	/**
	 * @param {Number} the time to evaluate against the VMAPParser.
	 * @return The amount of time remaining in the preroll, midroll or postroll.
	 */
	getAdSlotTimeRemaining: function(currentTime) {
		if (_.isString(currentTime)) {
			currentTime = this.rawTime(currentTime);
		}
		currentTime = Math.round(currentTime);
		var result = NaN;
		_.some(this.ranges, function(range) {
			if (_.contains(range, currentTime)) {
				result = _.last(range) - currentTime + 1;
				return true;
			}
		});
		return result;
	}
};
