/* exported PlaylistItem */
var PlaylistItem = function(options) {
	/**
	 * @ignore
	 * The src to play.
	 */
	this.src = options.src;
	this.isLive = options.isLive;
	this.index = options.index;
	/**
	 * @ignore
	 * Contains the rss metadata, and a mediaGen, which could contain the vmap.
	 */
	this.rss = {
		guid: options.uri,
		mediaGen: {}
	};
};

PlaylistItem.prototype = {
	/**
	 * @ignore
	 * If the item is in the process of being loaded. Currently, not used.
	 */
	loading: false,
	/**
	 * @ignore
	 * The item has been prepared.
	 */
	ready: false,
	/**
	 * @ignore
	 * The item is live.
	 */
	isLive: false
};
