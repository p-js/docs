/* exported PlayState */
/**
 * @class PJS.PlayState
 * When a {@link PJS.Events#stateChange} event is fired, the event's data property will be equal to one of these play states.
 * At the moment, there may be incongruities between html5 and flash state sequences.
 * Flash also has "initializing" and "connecting" states, which aren't available in the html5 player.
 */
var PlayState = {
	/**
	 * The player begins looking for media data.
	 * @property
	 */
	INITIALIZING: "loadstart",
	/**
	 * The video is playing.
	 * @property
	 */
	PLAYING: "playing",
	/**
	 * The video is paused.
	 * @property
	 */
	PAUSED: "paused",
	/**
	 * The video is seeking.
	 * @property
	 */
	SEEKING: "seeking",
	/**
	 * The video is stopped.
	 * @property
	 */
	STOPPED: "stopped",
	/**
	 * A stalled event fired.
	 * @property
	 */
	STALLED: "stalled",
	/**
	 * An error event fired.
	 * @property
	 */
	ERROR: "error"
};
