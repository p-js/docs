/* exported Managers */
var Managers = {
	chooseType: function(config) {
		var playlistManager;
		if (config.isVMAP()) {
			playlistManager = VMAPManagers.get();
		} else {
			playlistManager = MRSSManagers.get();
		}
		return Managers.get(playlistManager);
	},
	get: function(otherManagers) {
		var managers = [
			UserManager,
			DialogManager,
			PlaceholderManager,
			ActivatedVideoElementManager,
			PlaybackManager,
			APIManager,
			CorrectiveSeekManager,
			LoadingIconManager,
			BigPlayButtonManager,
			SimpleCCManager,
			VTTManager,
			OverlayManager,
			ClosedCaptionManager,
			AdDisplayManager,
			ControlsManager,
			TopPanelManager,
			FullScreenManager,
			ReportingManager,
			ConvivaManager
		];
		_.each(otherManagers, function(m) {
			managers.splice(m.index, 0, m.manager);
		});
		return managers;
	}
};
