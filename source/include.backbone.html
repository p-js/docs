/* global _, $, provide */
/* exported Backbone */
// change "this" to a custom scope that has _ and $.
var Backbone = (function() {
	//= ../../../components/backbone/backbone.js
	Backbone = this.Backbone;
	Backbone.$ = this.$;
	return Backbone;
}).apply({
	_: _,
	$: $
});
provide("backbone", Backbone);
