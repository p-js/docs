/* global _, Template */
/* exported TimedTextModel */
var TimedTextModel = {
	hasTracks: function(config) {
		return !_.isEmpty(config.timedText);
	},
	getCurrentTrack: function(config, lang) {
		var timedText = config.timedText;
		if (_.isEmpty(timedText)) {
			return false;
		}
		// search for timed text based on language.
		var foundTrack = _.findWhere(timedText, {
			lang: lang
		});
		// if none found, use the first
		if (!foundTrack) {
			foundTrack = _.first(timedText);
		}
		if (!foundTrack) {
			return null;
		}
		foundTrack.src = Template.process(foundTrack.src, config);
		// apply some defaults.
		return _.defaults(_.clone(foundTrack), {
			type: "application/ttml+xml",
			kind: "subtitles",
			lang: "en"
		});
	}
};
