/* global PJS, context, _, $*/
/**
 * @member PJS
 * @property {Boolean}
 * Set to true after the API is loaded.
 */
PJS.isReady = true;
(function() {
	// execute any on API callbacks.
	if (_.isFunction(PJS.onAPIReady)) {
		PJS.onAPIReady(PJS);
	}
	// ensure any legacy callbacks work
	var callbacks = context._pjsReady || [],
		legacyCallbacks = context._mtvnPlayerAPIReady || [];
	callbacks = callbacks.concat(legacyCallbacks);
	// simpler than the callback chain.
	_.each(callbacks, function(cb) {
		if (_.isFunction(cb)) {
			cb(PJS);
		}
	});
	// define the array, so we can override push.
	context._pjsReady = [];
	// override push to fire callbacks immediately
	context._pjsReady.push = function(cb) {
		cb(PJS);
	};
	// legacy callbacks
	context._mtvnPlayerAPIReady = context._pjsReady;
})();
/**
 * @ignore
 * allow auto creation of players by defining a data attribute.
 */
$(function() {
	$(".MTVNPlayer").addClass("pjs");
	$(".pjs[data-auto-create]").player();
});
