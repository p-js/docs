/* exported PlayheadOffsetManager */
/* global Module, _, Events*/
/**
 * @ignore
 * Manage the playhead offset when we're in full episode mode.
 * The playlist playhead will equal the offset plus the current playhead
 * Ensure that playheadOffset is reset on ads and on index change
 */
var PlayheadOffsetManager = Module.extend({
	playheadOffset: 0,
	initialize: function() {
		var player = this.options.player;
		_.bindAll(this, "onIndexChange", "getPlayhead", "onMetadata");
		if (this.config.useSegmentedScrubber()) {
			player.on(Events.INDEX_CHANGE, this.onIndexChange);
			player.on(Events.METADATA, this.onMetadata);
		}
	},
	onMetadata: function(event) {
		if (event.data.isAd) {
			this.playheadOffset = 0;
		}
	},
	onIndexChange: function(event) {
		var player = this.options.player,
			currentIndex = event.data;
		this.playheadOffset = 0;
		_.each(_.first(player.playlistMetadata.items, currentIndex), function(item) {
			this.playheadOffset += item.duration;
		}, this);
	},
	getPlayhead: function() {
		return this.playheadOffset + (this.player.playhead || 0);
	}
});
